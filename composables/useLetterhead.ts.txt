export const useLetterhead = async (_id: string) => {
	const {data: letterhead, pending} = await useFetch(`/api/settings/company/${_id}?h=2&w=10&b64=true&ext=.svg`, {
		method: "get",
		body: {collection: "configurations", projection: {property_type: 1, code: 1, name: 1}, filter: {}},
	});
	//console.log("letterhead", letterhead);
	return {letterhead, pending};
};
